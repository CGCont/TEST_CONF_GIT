-----------Configuración
git --version
git config --global user.name "Gerardo Contreras"
git config --global user.email cgcontreras23@gmail.com
git config --global core.editor "code --wait"
git config --global -e
git config --global core.autocrlf true //windows
git config --global core.autocrlf input //mac o linux
git config -h //lista de comando de git

---------Comandos de directorios
ls listar //directorios
pwd ubicar //directorio actual
cd directorio //cambiar de directorio
mkdir //crear carpeta
rm //eliminar archivo
mv mover o cambiar nombre de archivo
cat //mostrar contenido de archivo


----------comandos de git
git init //inicializar seguimiento
code . //abrir editor de texto en la ubicación actual
git status //conocer si los archivos tienen seguimiento
git status -s
git add . //agregar archivos al seguimiento
git commit -m "mensaje"
git restore --staged archivo //deshacer git add
git restore //restaurar archivo eliminado con git rm
git diff // visualizar cambios mas a detalle salir de la pantalla presioando q
git diff --staged
git log //logs de git
git log --oneline

---------- comandos para subir al server
git branch
git chekout -b nombre_de_la_rema
git merge nombre_rama //combinar ramas
git remote add origin https://github.com/CGCont/TEST.git
git push -u origin main
